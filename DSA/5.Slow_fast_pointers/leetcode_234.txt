class Solution {
    public boolean isPalindrome(ListNode head) {
        // [1 -> 2 -> 2 -> 1]        
        // [1 -> 2]
        StringBuilder sb = new StringBuilder();
        while(head != null) {
            sb.append(head.val);
            head = head.next;
        }
        int i=0, j=sb.length()-1;
        boolean ans = true;
        while(i<=j) {
            if(sb.charAt(i) != sb.charAt(j)) ans = false;
            i++;j--;
        }
        return ans;
    }
}
